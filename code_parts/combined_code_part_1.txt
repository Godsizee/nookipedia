<?php

namespace App\Core;

use PDO;
use PDOException;

class Database {
    private static $instance = null;
    private $connection;

    private function __construct() {
        # Laden der Umgebungsvariablen (Einfaches Mapping für dieses Beispiel)
        # In größeren Projekten würde man hier einen .env Parser nutzen.
        $env = parse_ini_file(__DIR__ . '/../../.env');

        $host = $env['DB_HOST'];
        $port = $env['DB_PORT'];
        $db   = $env['DB_NAME'];
        $user = $env['DB_USER'];
        $pass = $env['DB_PASS'];

        $dsn = "pgsql:host=$host;port=$port;dbname=$db";
        
        $options = [
            PDO::ATTR_ERRMODE            => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES   => false,
        ];

        try {
            $this->connection = new PDO($dsn, $user, $pass, $options);
        } catch (PDOException $e) {
            die("Datenbankverbindung fehlgeschlagen: " . $e->getMessage());
        }
    }

    public static function getInstance() {
        if (self::$instance === null) {
            self::$instance = new self();
        }
        return self::$instance;
    }

    public function getConnection() {
        return $this->connection;
    }
}

<?php

namespace App\Core;

class Router {
    protected $routes = [];

    public function add($url, $controller, $action) {
        $this->routes[$url] = [
            'controller' => $controller,
            'action' => $action
        ];
    }

    public function dispatch($url) {
        // Simple Cleaning
        $url = trim($url, '/');
        $url = explode('?', $url)[0];

        if (array_key_exists($url, $this->routes)) {
            $controllerName = "App\\Controllers\\" . $this->routes[$url]['controller'];
            $action = $this->routes[$url]['action'];

            if (class_exists($controllerName)) {
                $controller = new $controllerName();
                if (method_exists($controller, $action)) {
                    $controller->$action();
                } else {
                    $this->abort(404);
                }
            } else {
                $this->abort(404);
            }
        } else {
            // Standard-Route (Home) falls nichts gefunden wird oder 404
            $this->abort(404);
        }
    }

    protected function abort($code = 404) {
        http_response_code($code);
        require __DIR__ . "/../../views/errors/$code.php";
        die();
    }
}

# Autoloading (PSR-4 manuell für KISS)
spl_autoload_register(function ($class) {
    $prefix = 'App\\';
    $base_dir = __DIR__ . '/../app/';

    $len = strlen($prefix);
    if (strncmp($prefix, $class, $len) !== 0) {
        return;
    }

    $relative_class = substr($class, $len);
    $file = $base_dir . str_replace('\\', '/', $relative_class) . '.php';

    if (file_exists($file)) {
        require $file;
    }
});

use App\Core\Router;

# Router initialisieren
$router = new Router();

# Routen definieren

$router->add('insekten', 'CreatureController', 'insects');
$router->add('fische', 'CreatureController', 'fish');
$router->add('meerestiere', 'CreatureController', 'sea');

# Pfad anpassen, falls die App in einem Unterordner liegt

$requestUri = str_replace($scriptName, '', $requestUri);

$router->dispatch($requestUri);
